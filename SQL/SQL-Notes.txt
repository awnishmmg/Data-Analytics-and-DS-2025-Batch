select emp_id,emp_name,dept_name,salary from emp;

   select * from emp;

   select * from emp where emp_name = 'Awnish'

   -- How to find NULL and Not null 
     select * from emp where salary = null -- invalid
     select * from emp where salary != null  -- Invalid 
     
     -- for null we use is null 
     select * from emp where salary is null -- invalid
     select * from emp where salary is not null  -- Invalid 
     
    -- in and not in 
    -- in and not in is used when you want the filter the data on the basis 
    -- the given list of items 
    -- Note for list of item we use ()
    -- id in (11,15,19,13)   
     
     select * from emp where emp_id in (11,15,19,13);
     select * from emp where emp_id not in (11,15,19,13);
    
     -- How to include Null with in
     select * from emp where emp_id in (11,15,19,13,NULL);
     select * from emp where salary in (1000,2000,NULL); -- Invalid 
     -- null should be used as or
     select * from emp where salary in (1000,2000) or salary is null;
     select * from emp where salary not in (1000,2000) or salary is not null;
    
    -- Between Operator : we want to find the data in some given Range 
    select * from emp where salary between  2000 and 7000;
    -- Inserting the Record to match the Range b/w 2000 to 7000
    insert into emp(emp_name,dept_name,salary) values
    ('Shivi','CEO',3000),
    ('Kivi','CTO',4000);
   
    -- How to select the range without Between Operator 
     select * from emp where salary >= 2000 and salary <= 7000;
    
    -- How to combine multiple columns using and and or 
    -- using or 
    -- select those emp where department sales or HR also salary is < 5000
    select * from emp where dept_name in ('HR','SALES') or salary < 5000;
    select * from emp where dept_name in ('HR','SALES') and salary < 5000;
  
   -- Searching or like operator 
   -- wild character
   -- * : All Columns 
   -- % (anything) and - (1) digit match exactly any as per digit 
   
   -- pre and post 
   -- A% : must start from A but could be anything from A-z 
   -- A--- : must start from A and there should be exactly 4 digit and could 
   -- aything 
   -- %A : starting could be anything but must end by A 
   
   select * from emp;
   select * from emp where emp_name like 'am%';
   select * from emp where emp_name like '%i';
   select * from emp where emp_name like '%am%';
   
  select * from emp where emp_name like 'A%';
  select * from emp where emp_name like 'A___';
  select * from emp where emp_name like 'A___%';
